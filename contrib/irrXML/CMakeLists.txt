set( IrrXML_SRCS
  CXMLReaderImpl.h
  heapsort.h
  irrArray.h
  irrString.h
  irrTypes.h
  irrXML.cpp
  irrXML.h
)

  ARCHIVE DESTINATION ${ASSIMP_LIB_INSTALL_DIR}
